include(srcs.cmake)

foreach (isource ${ARCANE_SOURCES})
  set(_FILE ${ARCANECEA_SRC_PATH}/arcane/cea/${isource})
  string(REGEX MATCH "\.(h|H)$" _OUT_HEADER ${isource})
  if (_OUT_HEADER)
    get_filename_component(_HEADER_RELATIVE_PATH ${isource} DIRECTORY)
    install(FILES ${_FILE} DESTINATION include/arcane/cea/${_HEADER_RELATIVE_PATH})
  endif ()
endforeach ()
